内置内建元素

这些就是 Gtk 组件，无需显式导入即可使用。例如 和 是完全相同的。

Gtk4

box

Gtk.Box

<box orientation={Gtk.Orientation.HORIZONTAL}>
  <Child />
  <Child />
  <Child />
</box>

button

Gtk.Button

<button onClicked={() => print("已点击")}>
  <Child />
</button>

centerbox

Gtk.CenterBox

<centerbox orientation={Gtk.Orientation.HORIZONTAL}>
  <Child $type="start" />
  <Child $type="center" />
  <Child $type="end" />
</centerbox>

drawingarea

Gtk.DrawingArea

<drawingarea
  $={(self) =>
    self.set_draw_func((area, cr, width, height) => {
      //
    })
  }
/>

entry

Gtk.Entry

<entry
  placeholderText="开始输入..."
  text=""
  onNotifyText={({ text }) => print(text)}
/>

image

Gtk.Image

<image
  file="/path/to/file.png"
  iconName="system-search-symbolic"
  pixelSize={16}
/>

label

Gtk.Label

<label
  label="<span foreground='blue'>文本</span>"
  useMarkup
  wrap
  ellipsize={Pango.EllipsizeMode.END}
/>

levelbar

Gtk.LevelBar

<levelbar
  orientation={Gtk.Orientation.HORIZONTAL}
  widthRequest={200}
  value={0.5}
/>

menubutton

Gtk.MenuButton

<menubutton>
  按钮内容
  <popover>弹出内容</popover>
</menubutton>

overlay

Gtk.Overlay

<overlay>
  <Child />
  <Child $type="overlay" />
  <Child $type="overlay" />
</overlay>

revealer

Gtk.Revealer

<revealer
  transitionType={Gtk.RevealerTransitionType.SLIDE_RIGHT}
  revealChild={true}
  onNotifyChildRevealed={() => print("动画完成")}
>
  <Child />
</revealer>

scrolledwindow

Gtk.ScrolledWindow

<scrolledwindow maxContentHeight={500}>
  <Child />
</scrolledwindow>

slider

Astal.Slider

<slider
  value={0.5}
  min={0}
  max={1}
  onChangeValue={({ value }) => print(value)}
/>

stack

Gtk.Stack

<stack visibleChildName="child2">
  <Child $type="named" name="child1" />
  <Child $type="named" name="child2" />
</stack>

switch

Gtk.Switch

<switch active={true} onNotifyActive={({ active }) => print(active)} />

togglebutton

Gtk.ToggleButton

<togglebutton active={true} onToggled={({ active }) => print(active)} />

window

Astal.Window

<window
  visible
  namespace="bar"
  class="Bar"
  monitor={0}
  exclusivity={Astal.Exclusivity.EXCLUSIVE}
  keymode={Astal.Keymode.ON_DEMAND}
  anchor={
    Astal.WindowAnchor.TOP | Astal.WindowAnchor.LEFT | Astal.WindowAnchor.RIGHT
  }
/>

WARNING

在 Gtk4 中，窗口默认不可见，所以不要忘记传递 visible。
